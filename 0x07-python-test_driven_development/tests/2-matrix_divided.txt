# Import the matrix_divided function
>>> from 2-matrix_divided import matrix_divided

# Test case 1: Divide matrix elements by 3
>>> matrix = [
...     [1, 2, 3],
...     [4, 5, 6]
... ]
>>> result = matrix_divided(matrix, 3)
>>> result  # Check the result
[[0.33, 0.67, 1.0], [1.33, 1.67, 2.0]]

# Test case 2: Divide matrix elements by 2
>>> matrix = [
...     [10, 20],
...     [30, 40],
...     [50, 60]
... ]
>>> result = matrix_divided(matrix, 2)
>>> result  # Check the result
[[5.0, 10.0], [15.0, 20.0], [25.0, 30.0]]

# Test case 3: Divide matrix elements by -2
>>> matrix = [
...     [-1, -2, -3],
...     [-4, -5, -6]
... ]
>>> result = matrix_divided(matrix, -2)
>>> result  # Check the result
[[0.5, 1.0, 1.5], [2.0, 2.5, 3.0]]

...     [1, 2, 3],

...     result = matrix_divided(matrix, 0)
... except ZeroDivisionError as e:
...     str(e)  # Check the error message

...     [1, 2, '3'],
...     [4, 5, 6]
